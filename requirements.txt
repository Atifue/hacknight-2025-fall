#   brew install ffmpeg

# For wav2lip (optional - only if using video lipsync):
#   brew install git-lfs
#   git lfs install
#   git lfs pull

# Python setup:
#   python3 -m venv revoice-env
#   source revoice-env/bin/activate
#   pip3 install --upgrade pip
#   pip3 install -r requirements.txt

# For full lipsync feature (optional):
#   pip3 install onnx onnxruntime scipy opencv-python insightface

# API Keys:
#   Create detection-Files/.env with:
#   GEMINI_API_KEY=your-key
#   ELEVENLABS_API_KEY=your-key
